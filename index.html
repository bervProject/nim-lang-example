# Maximum Number

let x = [1, 4, 2, 3, 5, 8, 9, 7, 6, 0]
var max = x[0]
for i in 1..high(x):
  if max < x[i]:
    max = x[i]

echo "Nilai maksimal : ", max

# Swap Value

var x, y, temp: int
x = 3
y = 2
# show initial value
echo "x ", x
echo "y ", y
# swaping
temp = x
x = y
y = temp
# show swap value
echo "x ", x
echo "y ", y
